<script>
// Beispiel-Daten (Simulation)
const data = {
  klausuren: [
    { datum: "15.04.2026", fach: "Mathematik", zeit: "08:00 - 10:30", raum: "Aula" },
    { datum: "17.04.2026", fach: "Deutsch", zeit: "09:00 - 11:15", raum: "204" },
    { datum: "20.04.2026", fach: "Englisch", zeit: "10:00 - 12:30", raum: "Aula" }
  ],
  transport: [
    { linie: "SB91", richtung: "Stadtbus", abfahrt: "in 4 Min" },
    { linie: "RE19", richtung: "Düsseldorf Hbf", abfahrt: "in 12 Min" }
  ],
  news: [
    { datum: "12.03.", titel: "Sportfest", text: "Anmeldung ab sofort im Sekretariat." }
  ]
};

// Render-Funktionen
function render() {
  document.getElementById('klausur-table').innerHTML =
    data.klausuren.map(k => `
      <tr>
        <td>${k.datum}</td>
        <td>${k.fach}</td>
        <td>${k.zeit}</td>
        <td>${k.raum}</td>
      </tr>
    `).join("");

  document.getElementById('transport-table').innerHTML =
    data.transport.map(t => `
      <tr>
        <td>${t.linie}</td>
        <td>${t.richtung}</td>
        <td>${t.abfahrt}</td>
      </tr>
    `).join("");

  document.getElementById('news-table').innerHTML =
    data.news.map(n => `
      <tr>
        <td>${n.datum}</td>
        <td>${n.titel}</td>
        <td>${n.text}</td>
      </tr>
    `).join("");
}

// Simulierte KI-Analyse
function simulateAI() {
  document.getElementById("status").innerText = "KI analysiert Dokument...";
  setTimeout(() => {
    document.getElementById("status").innerText = "✅ Daten erfolgreich erkannt!";
    render();
  }, 1500);
}
</script>
